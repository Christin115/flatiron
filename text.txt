START PROGRAM

CREATE user savings goal

INPUT user_id
INPUT savings_goal
INPUT target_amount
INPUT target_date
INPUT amount_saved

STORE user_id
STORE savings_goal
STORE target_amount
STORE target_date
STORE amount_saved

DISPLAY "Savings goal created successfully"

WHILE amount_saved < target_amount

    DISPLAY "Enter updated savings amount"
    INPUT deposit_amount

    UPDATE amount_saved = amount_saved + deposit_amount

    STORE updated amount_saved

    CALCULATE remaining_amount = target_amount - amount_saved

    RETURN remaining_amount

    DISPLAY "Current Savings:", amount_saved
    DISPLAY "Remaining Amount:", remaining_amount

    IF amount_saved >= target_amount THEN

        SET goal_status = "Completed"

        RETURN goal_status

        COMPOSE completion message

        SEND notification:
        "Congratulations! Your savings goal has been reached."

    ELSE

        SET goal_status = "In Progress"

        RETRIEVE current_date

        CALCULATE days_remaining = target_date - current_date

        IF days_remaining <= 7 THEN

            COMPOSE reminder message

            SEND notification:
            "Reminder: Your savings goal deadline is approaching."

        ENDIF

    ENDIF

ENDWHILE

DISPLAY "Goal tracking complete"

END PROGRAM
